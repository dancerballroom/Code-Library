Populate_Test_Data.bat

@Echo Off
setlocal enabledelayedexpansion

rmdir /S /Q SWAP

If NOT exist "SWAP" ( 
    mkdir "SWAP" 
    echo Folder created.
) else ( 
    echo Folder already exists! 
) 

echo P > P.txt
pushd SWAP
for /l %%x in (1, 1, 12) do (
    if %%x LEQ 9 SET Zeroes=00
    if %%x GEQ 10 SET Zeroes=0
    if %%x GEQ 100 SET Zeroes=

    echo %%x > "!Zeroes!%%x - P.txt"    
)
popd

Swap.bat (Main Program)

@Echo Off
setlocal enabledelayedexpansion

REM Create New File From %1 variable
SET SWAP_DIR=SWAP

REM Copy First-File to Temp file

REM COMMENT FOR NEXT TIME
REM This happens whether or not %1 = Something

echo MOVE "P.txt" "%SWAP_DIR%\temp.txt"
MOVE "P.txt" "%SWAP_DIR%\temp.txt"

pushd SWAP

REM Count Backup Files
for /f %%A in ('dir *P.* ^| find "File(s)"') do set cnt=%%A

REM echo Calculate Zeroes Padding for Filenames
if (%cnt% LEQ 9) ( SET Zeroes=00)
if (%cnt% GEQ 10) ( SET Zeroes=0)
if (%cnt% GEQ 100) ( SET Zeroes=)

popd

REM COMMENT FOR NEXT TIME
REM PseudoCode
REM if %1 = Something, P.txt = %1, no shuffle 2-end, add new node
REM else P.txt = 1st node, shuffle, and P.txt -> temp
REM NEXT TIME
REM ========== %1 = NOTHING -----------

REm Copy Next-Up to First-File
Echo MOVE "%SWAP_DIR%\%Zeroes%1 - P.txt" P.txt
MOVE "%SWAP_DIR%\%Zeroes%1 - P.txt" P.txt

pushd SWAP

REM RE-Count Files
for /f %%A in ('dir *P.* ^| find "File(s)"') do set cnt=%%A

REM Move the Queue from next-to-last to Next-Up
for /l %%x in (2, 1, %cnt%) do (

    set /a cnt3+=1
    if !cnt3! LEQ 9 SET Zeroes=00
    if !cnt3! GEQ 10 SET Zeroes=0
    if !cnt3! GEQ 100 SET Zeroes=

    if %%x LEQ 9 SET Zeroes1=00
    if %%x GEQ 10 SET Zeroes1=0
    if %%x GEQ 100 SET Zeroes1=

    Echo MOVE "!Zeroes1!%%x - P.txt" "!Zeroes!!cnt3! - P.txt"
    MOVE "!Zeroes1!%%x - P.txt" "!Zeroes!!cnt3! - P.txt"
)
set /a cnt3+=1
if !cnt3! LEQ 9 SET Zeroes=00
if !cnt3! GEQ 10 SET Zeroes=0
if !cnt3! GEQ 100 SET Zeroes=

if %%x LEQ 9 SET Zeroes1=00
if %%x GEQ 10 SET Zeroes1=0
if %%x GEQ 100 SET Zeroes1=

REM Copy First-File to Last-File 
REM MOVE "%SWAP_DIR%\temp.txt" "%SWAP_DIR%\%Zeroes%%cnt3% - P.txt"
ECHO MOVE "temp.txt" "%Zeroes%%cnt3% - P.txt"
MOVE "temp.txt" "%Zeroes%%cnt3% - P.txt"

REM COMMENT FOR NEXT TIME
REM =========== END of %1 == Nothing =============
REM ------------ %1 = Something ------------
REM MOVE "%1" P.txt
REM SWAP\temp.txt receives next number in series

cls
Echo Current : 
Type ..\P.txt
Echo Archived : 
REM View Contents of each File
for /l %%x in (1, 1, %cnt%) do (
    if %%x LEQ 9 SET Zeroes=00
    if %%x GEQ 10 SET Zeroes=0
    if %%x GEQ 100 SET Zeroes=

    REM Echo Type "!Zeroes!%%x - P.txt"
    Type "!Zeroes!%%x - P.txt"
)
popd

pause

Add_Playlist.bat

@Echo Off
setlocal enabledelayedexpansion

pushd SWAP

REM Count Backup Files
for /f %%A in ('dir *P.* ^| find "File(s)"') do set cnt=%%A

REM Set Next Number
SET /A cnt+=1

REM Leading Zeroes 
if %cnt% LEQ 9 SET Zeroes=00
if %cnt% GEQ 10 SET Zeroes=0
if %cnt% GEQ 100 SET Zeroes=

echo %cnt% > "!Zeroes!%cnt% - P.txt"

popd
