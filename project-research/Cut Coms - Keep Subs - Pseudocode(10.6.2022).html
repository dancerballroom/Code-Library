5. Simple FFMPeg Join ALL MP4

cd "%dirpath%"
for %%a in (Chapter*.mp4) do echo file '%%a' >> Concat_List.txt
ffmpeg -f concat -safe 0 -i Concat_List.txt 
        -c copy -scodec copy "Output\$VideoNameNoExtension.mp4"

# MOVIE.srt: An Index Array
$SRT_Index = $Content | foreach { If ($_ -match "(?<=(?:[\r\n]|^))\d+") { $Matches[0] } }
# MOVIE.srt: An Array of The Start Time Signature of each Subtitle
$SRT_TimeSigs_START = $Content | foreach { If ($_ -match "^(\d){2}:(\d){2}:(\d){2},(\d){3}") { $Matches[0] } }
# MOVIE.srt: An Array of The End Time Signature of each Subtitle
$SRT_TimeSigs_END = $Content | foreach { If ($_ -match "(\d){2}:(\d){2}:(\d){2},(\d){3}$") { $Matches[0] } }

$SubsArray = $Content  | foreach { If ($_ -match "(?<Cue>(?:.|\s){0,}?(?=\z|(?<=[\r\n])\d+))") { $Matches[0] } }
