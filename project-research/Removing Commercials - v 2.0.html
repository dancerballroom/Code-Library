# NOTE:  Batchfiles show up in the SendTo Directory, but PowerScript files do not.

@ECHO OFF

REM C:\DIR\DIR\DIR\FIle.mp4
SET arg1=%~1             

REM FIle    
SET arg2=%~n1

REM FIle.mp4
SET arg3=%~n1%~x1

REM C:\DIR\DIR\DIR\
SET arg4=%~d1%~p1

SET backup=%~d1%~p1\backup

SET PSScript=C:\Users\Larry Bear\Desktop\Power Shell Research\FINAL_Version_RECEIVER.ps1
SET PowerShellDir=C:\Windows\System32\WindowsPowerShell\v1.0
CD /D "%PowerShellDir%"
Powershell -Command "& '%PSScript%' '%arg1%' '%arg2%'"

CD /D "C:\ScriptTest"
REM # Create an Empty File
copy NUL Concat_List.txt >NUL

REM # Create a listing of all Chapter files in the current directory (No Commercials)
for %%a in (Chapter*) do echo file '%%a' >> Concat_List.txt

REM # Run the ffmpeg command to concatenate the file
REM # Would not run with out the -safe 0 command
ffmpeg -f concat -safe 0 -i Concat_List.txt -c copy "%arg3%"
del Chapter*.mp4
del Concat_List.txt
if not exist "%backup%" mkdir "%backup%" 
MOVE "%arg1%" "%backup%"
Move "%arg3%" "%arg4%"

echo Done!
EXIT /B

# The first parameter sent to the file is the fully qualified Path and Filename
$arg1=$args[0]
# The second parameter is the Filename and Extension
$arg2=$args[1]

#Create TEMP folders
$Path = "C:\POWER-SHELL-SCRIPTS"
# $Commies = "C:\POWER-SHELL-SCRIPTS\Commercials"
if (Test-Path $Path) {   
    Write-Host "Folder Exists"
}
else
{
    #PowerShell Create directory if not exists
    New-Item -ItemType Directory -Path $Path
    Write-Host "Folder Created successfully"
}

# Make Sure the Path here is to the folder with ffmpeg.exe and ffprobe.exe
cd "C:\FFMPEG\bin" #"

$videoIncludingPath = $arg1
$VideoNameNoExtension = $arg2

#Get Chapters from file, Convert from json
# Next on One Line
$json = (.\ffprobe.exe -i "$videoIncludingPath" -show_chapters -print_format json) | Out-String | ConvertFrom-Json

#Output only endtimes to varible
$Endtimes = $json.chapters.end_time

# LAST SECTION: Create the Chapter MP4s
#First start time is 0 seconds
$Starttime = 0.000000

#For each endtime output file
for ($i=0; $i -lt $Endtimes.length; $i+=2) {
    $ChapterNum = $i + 1
    if ($ChapterNum  -le 9) {
     #  Next on One Line
    .\ffmpeg -i "$videoIncludingPath" -ss $Starttime -to $Endtimes[$i] "$path\Chapter 0$chapterNum - $VideoNameNoExtension.mp4"
    } Else {
     #   Next on One Line
    .\ffmpeg -i "$videoIncludingPath" -ss $Starttime -to $Endtimes[$i] "$path\Chapter $chapterNum - $VideoNameNoExtension.mp4"
    }
    $Starttime = $Endtimes[$i+1]
}
