Passing_String_Arrays_to_Functions.ps1

Set-Alias ds Write-Host
Set-Alias rh Read-Host

$Continue_Til_End = 1

while ($Continue_Til_End) {
   # Main Menu
    ds "Run_Script = 1 to Run Program"
    ds "           = 2 to Debug"
    ds "           = 3 to Exit"

    $Choice = rh "(1 Run, 2 Debug, 3 Exit)   "

    If ($Choice -eq 1) { $Run_Script = 1 } else { $Run_Script = 0 }
    If ($Choice -eq 3) { Exit }

    Function Run_Program([string[]]$some_string_array) {
        ds "---->You must be running!"
        foreach ($x in $some_string_array)
        {
            # Print Each String Array item
            ds $x
        }
    }

    Function Debug_Program ([string[]]$some_string_array) {
        ds "You are Sending in these Variables:"
        foreach ($x in $some_string_array)
        {
            # Print Each String Array item
            ds `$`x = $x
        }
    }

# Run the Script -> $Run_Script = 1
    if ($Run_Script) {
        Run_Program 1, 2, 3
        Run_Program 1 2, 3
        Run_Program 1, 2 3
    }

# Debug the Script -> $Run_Script = 0
    # Print the Calling Function and it's Array Parameters
    # Call Function with Array Parameters
    if (! $Run_Script) {
        ds "Debug_Program 1`, 2`, 3"
        Debug_Program 1, 2, 3
        ds "Debug_Program 1 2`, 3"
        Debug_Program 1 2, 3
        ds "Debug_Program 1`, 2 3"
        Debug_Program 1, 2 3
    }
}
