DEBUG.ps1 - Version 1.1 (More Code Expandable than  v1.0 below)

Set-Alias ds Write-Host
Set-Alias rh Read-Host

$Debug = ""
$Run_Script = 0
while ($Debug -eq "") {
    ds "1 Debug Power Script"
    ds "2 Run Power Script`n"
    
    $Debug= rh "Choice "
    
    if ($Debug -eq 1) {
            ds "`nDebug Power Script"
    } elseif ($Debug -eq 2) {
            ds "`nRun Power Script"
            $Run_Script=1   
    } else {
            ds "`nInvalid Entry`n"
            ds "Terminating Script`n"
            Exit
    }
}

if ($Run_Script) {
    # All Run Code Runs Here 
    ds "-->Running Power Script"
} else {
    # All Debug Code Runs Here
    ds "-->Debugging Power Script"
}
ds ""
if (! $Run_Script) {
    # More Debugging Code using the NOT operator ! instead of ELSE above
    ds "---->You must be debugging!"
}
