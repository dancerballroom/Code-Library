Converting Movie Bookmarks to TXT/JSON - Precision Subtitle Editing out Commercials

Process for Normal (Not Broken) MP4 to set up Automatic SubtitleEdit MARKERS.

  • After Chopping Normal.MP4 Front and Back (CC-CHOP-CHOP...BAT)
  • Drag-Drop Normal.MP4 with Pattern SHOW-COMMS-SHOW-COMMS (ODD Show Placement:  1, 3, 5, 7) 
    • onto BKMK-2-JSON-ODD.
  • Drag-Drop Normal.txt onto TXT2JSON.bat
  • Double Click NORMAL.JSON
  • Check Each Commercial Start-End
  • Drag Start-End to Eliminate Commercial REMNANTS
  • Save NORMAL.JSON
  • Right-Click/SendTo/DROP-JSON-HERE.bat to Eliminate Commercials

BKMK-2-JSON-ODD.PS1


Set-Alias rh Read-Host
Set-Alias ds Write-Host

$path=$args[1]
$file_with_ext=$path + $args[0]
$filename=$args[2]
$file=$path + $filename
Write-Host "filename: $filename"
Write-Host "file: $file"
Write-Host "path: $path"

$index = "index"
$start = "start"
$end = "end"
$text = "text"
$comms = "Comms"

# Life Saver Function
function Q([String]$x)
{
  $result = "`"$x`""
  #` QUIP
  return $result
}
# 

# Path here is to the folder with ffmpeg.exe and ffprobe.exe
cd "C:\FFMPEG\bin" #"

#Get Chapters from file, Convert from json
$json = (.\ffprobe.exe -i "$file_with_ext" -show_chapters -print_format json) | Out-String | ConvertFrom-Json
#Output only endtimes to varible
$Endtimes = $json.chapters.end_time

$INFO = $INFO + "[`n" 

# Target Format: Laverne and Shirley - 0s08e17 - Ghost Story
#[
#{"index":2,"start":00:01:16:523,"end":00:02:02:130,"text":["Comms"]},
#{"index":4,"start":00:10:21:881,"end":00:11:36:922,"text":["Comms"]},
#{"index":6,"start":00:24:54:636,"end":00:25:09:829,"text":["Comms"]},
#]


for ($i=2; $i -lt $Endtimes.length; $i+=2) {

    $St = [timespan]::fromseconds($Endtimes[$i - 2])
    $Et = [timespan]::fromseconds($Endtimes[$i - 1])
    $INFO = $INFO + "{"
    $INFO = $INFO + (Q $index) + ":"
    $INFO = $INFO + $i + ","    
    $INFO = $INFO + (Q $start) + ":"
    $INFO = $INFO + $St.ToString("hh\:mm\:ss\:fff") + ","
    $INFO = $INFO + (Q $end) + ":"
    $INFO = $INFO + $Et.ToString("hh\:mm\:ss\:fff") + ","
    $INFO = $INFO + (Q $text)    
    $INFO = $INFO + ":[" + (Q $comms) + "]},"
    $INFO = $INFO + "`n"
 }

$INFO = $INFO + "]" 
$Infofile = "$file.txt"

$Stream = [System.IO.StreamWriter]::new($Infofile)
$Stream.WriteLine($INFO)
$Stream.Close()
