# Script Exits instead of Returning to the Menu 

CC_Times_UP.bat

@Echo Off
setlocal enableDelayedExpansion

:: The One Variable that needs to be changed on Every Computer
:: The Path to the SendTo based on the Current USER Directory
pushd C:\Users\dance\AppData\Roaming\Microsoft\Windows\SendTo

:RERUN
cls
echo Running !TimesToRun! Times
echo.
:: # Show the current results for CC files
dir /b CC*.bat
echo.

:: # Show USER the Round Robin Segments
echo Forward Round Robin (Seconds to Cut off Video Start)
echo.
echo   01-13
echo   14-19
echo   20-29
echo   30-39
echo   40-49
echo   50-...
echo.
echo 0 = Exit
:: # Run the CC_UP_RoundRobin Batchfile TimesToRun Times 
set /p TimesToRun="Times to Run CC_UP_RoundRobin.bat -> (1-6) :> "
for /L %%a in (1,1,!TimesToRun!) Do echo %%a 

:choice
if "%TimesToRun%" EQU "0" goto :EOF
echo Running !TimesToRun! Times


:: # The Magic in One Line that makes everything work!
for /L %%a in (1,1,!TimesToRun!) Do Call CC_UP_RoundRobin.bat

goto :RERUN

:eof
echo.
echo This is the End!
echo Bye, Bye!
echo.

:: Pop Goes the Directory
popd
