Menu-Reg-Generator-V3.bat

@echo off
Setlocal enabledelayedexpansion
Set MG=Menu-Reg-Generator-V3.reg
Set MGDEL=Menu-Reg-Generator-V3-DELETE.reg
Set MGGraphic=Menu-Reg-Generator-Graphic-V3.txt

:: Check WMIC is available
WMIC.EXE Alias /? >NUL 2>&1 || GOTO s_error

:: Use WMIC to retrieve date and time
FOR /F "skip=1 tokens=1-6" %%G IN ('WMIC Path Win32_LocalTime Get Day^,Hour^,Minute^,Month^,Second^,Year /Format:table') DO (
   IF "%%~L"=="" goto s_done
      Set _yyyy=%%L
      Set _mm=00%%J
      Set _dd=00%%G
      Set _hour=00%%H
      Set _minute=00%%I
)
:s_done

:: Pad digits with leading zeros
      Set _mm=%_mm:~-2%
      Set _dd=%_dd:~-2%
      Set _hour=%_hour:~-2%
      Set _minute=%_minute:~-2%


:: Display the date/time in ISO 8601 format:
:: Set MyTime=%_yyyy%-%_mm%-%_dd%.%_hour%-%_minute%
:: Set MyDateTime=%_minute%-%_hour%_%_dd%-%_mm%-%_yyyy%
Set MyDateTime=%_hour%-%_minute%--%_mm%-%_dd%-%_yyyy%

Set MGCommand=COMMANDS-V3-%MyDateTime%.txt
REM Echo !MGCommand!

::Main Menu - Define Carriage Return
set CRLF=^


:top
::Main Menu
set prompt="0 = Exit!CRLF!1 = Sub-menu!CRLF!2 = command!CRLF!3 = Create REG-File!CRLF!: "

set /p menuitem=!prompt!
echo !menuitem! >> !MGCommand!
REM :: Menu Choices
REM :: 1 = Exit Program
IF "%menuitem%"=="0" (
    REM :: Exit-Condition
    REM :: 2 = Create Submenu
) ELSE IF "%menuitem%"=="1" (

set /p Submenu="Name of Sub-menu: "
echo !Submenu! >> !MGCommand!
set /p MenuNum="Menu Number: "
echo !MenuNum! >> !MGCommand!

REM Creation File
Echo.>> !MG!
Echo [HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!] >> !MG!
Echo "Icon"="shell32.dll,176" >> !MG!
Echo "MUIVerb"="!Submenu!" >> !MG!
Echo "SubCommands"="" >> !MG!
Echo.>> !MG!

REM Deletion File
Echo.>> !MGDEL!
Echo [-HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!] >> !MGDEL!
Echo "Icon"="shell32.dll,176" >> !MGDEL!
Echo "MUIVerb"="!Submenu!" >> !MGDEL!
Echo "SubCommands"="" >> !MGDEL!
Echo.>> !MGDEL!

REM :: To Show in Command Window ONLY
Echo [HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!]
Echo "Icon"="shell32.dll,176"
Echo "MUIVerb"="!Submenu!"
Echo "SubCommands"=""
Echo.

REM :: Graphic-V3
Echo ^|  ^\-- !Submenu! >> !MGGraphic!

goto :top

REM :: 2 = Create Command
) ELSE IF "%menuitem%"=="2" (
set /p NaniO="Give Me A Path: -> "
echo !NaniO! >> !MGCommand!
set N=!NaniO:\=\\\\!
set /p My-Bat="Name of Batch file: "
echo !My-Bat! >> !MGCommand!
set /p My-Command="Name of Command: "
echo !My-Command! >> !MGCommand!
set /p ItemNum="Item Number: "
echo !ItemNum! >> !MGCommand!

REM Creation File
Echo.>> !MG!
Echo [HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!\shell\item!ItemNum!] >> !MG!
Echo "Icon"="shell32.dll,208" >> !MG!
Echo "MUIVerb"="!My-Command!" >> !MG!
Echo.>> !MG!
Echo [HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!\shell\item!ItemNum!\command] >> !MG!
Echo @="!N!\\\\!My-Bat! \"%%1\"" >> !MG!
Echo.>> !MG!

REM Deletion File
Echo.>> !MGDEL!
Echo [-HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!\shell\item!ItemNum!] >> !MGDEL!
Echo "Icon"="shell32.dll,208" >> !MGDEL!
Echo "MUIVerb"="!My-Command!" >> !MGDEL!
Echo.>> !MGDEL!
Echo [-HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!\shell\item!ItemNum!\command] >> !MGDEL!
Echo @="!N!\\\\!My-Bat! \"%%1\"" >> !MGDEL!
Echo.>> !MGDEL!


REM :: To Show in Command Window ONLY
Echo.
Echo [HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!\shell\item!ItemNum!]
Echo "Icon"="shell32.dll,208"
Echo "MUIVerb"="!My-Bat!" 
Echo.
Echo [HKEY_CLASSES_ROOT\*\shell\%ROOT_DIR%\shell\menu!MenuNum!\shell\item!ItemNum!\command]
Echo @="!N!\\\\!My-Bat! \"%%1\""
Echo.

REM :: Graphic-V3
Echo ^|  ^|  !My-Bat! >> !MGGraphic!

goto :top

REM :: 3 = Create REG File 
) ELSE IF "%menuitem%"=="3" (

REM :: ROOT DIR-Name

set /p ROOT_DIR="Root Directory Name?  : "
REM Creation File
Echo Windows Registry Editor Version 5.00 > !MG! 
Echo. >> !MG!
Echo [HKEY_CLASSES_ROOT\*\shell\!ROOT_DIR!] >> !MG!
Echo "Icon"="shell32.dll,137" >> !MG!
Echo "Subcommands"="" >> !MG!
Echo.>> !MG!

REM Deletion File
Echo Windows Registry Editor Version 5.00 > !MGDEL! 
Echo. >> !MGDEL!
Echo [-HKEY_CLASSES_ROOT\*\shell\!ROOT_DIR!] >> !MGDEL!
Echo "Icon"="shell32.dll,137" >> !MGDEL!
Echo "Subcommands"="" >> !MGDEL!
Echo.>> !MGDEL!

REM :: To Show in Command Window ONLY
Echo Windows Registry Editor Version 5.00
Echo. 
Echo [HKEY_CLASSES_ROOT\*\shell\!ROOT_DIR!]
Echo "Icon"="shell32.dll,137"
Echo "Subcommands"=""
Echo. 

REM :: Graphic-V3
Echo \---------------------- > !MGGraphic!
Echo Menu-Reg-Generator-Graphic-V3.txt >> !MGGraphic!
Echo Creation Date >> !MGGraphic!
echo %DATE% at %TIME% >> !MGGraphic!

Echo. >> !MGGraphic!
Echo ^/-- !ROOT_DIR! >> !MGGraphic!

goto :top

REM :: Invalid Entry = Return to Main Menu
) ELSE (
  echo Try Again
  goto :top
)
REM :: Graphic-V3
Echo \---------------------- >> !MGGraphic!
Echo. >> !MGGraphic!
type Menu-Reg-Generator-Graphic-V3.txt
pause
Exit 0

POST  Program NOTES:

  • Similar to SendTo, but you control the Location and folder structure.
  • There seems to be a limitation to the number of items you can add through a reg file.  So far I think it’s 13 items.  I ran across this by accident, but am not able to find any documentation on this.
  • I default to adding passing a “%1” after each .bat, .vbs in the script.  This may need tweaking in the registry if there is no “%1” or “%1” is not necessary.
  • .VBS files need to be called using “wscript.exe“ ”VBS-FILE.vbs“
  • I have not tested this script with EXE’s, but theoretically, it should work.
