000_HTML_FILES_GEN-main.bat

@echo off
SETLOCAL ENABLEDELAYEDEXPANSION
REM  For Each HTML
REM    Create an Entry in main.html
REM 
REM  Exclude Files not to be processed
    set exclusionlist="index.html menu.html main_OLD.html menu.html main.html"
echo %~d0%~p0
    set atom_resources=%~d0%~p0resources\
    set cur_dir="%~d0%~p0"
    set default_menu=%~d0%~p0resources\menu.html

    set main_html="!cur_dir!main.html"

REM  Copy the Default Menu (left Frame)
echo copy "!default_menu!" .
    copy "!default_menu!" .

REM  START main.html
    type "!atom_resources!TOP-MAIN.txt" > !main_html!
    set /a count=0
    set CRLF=^

REM  Build main.html TABLE
REM  1st Table Row Start
    Echo ^<tr^>  >> !main_html!

REM  Process MIDDLE
For %%a In (*.html) Do Call :Sub "%%a"
goto :Finish
goto :ENDER

::-- getFirstXChars
:getFirstXChars
    set sValIn=%1
    set /a iNo=%2
    set vWorkVal=%%sValIn:~0,%iNo%%%
    call:getFirstX %vWorkVal% vWorkVal2
    set %3=%vWorkVal2%

:getFirstX
    set %2=%~1
goto:eof

:Sub
echo.!exclusionlist! | findstr /C:%1 1>nul

if errorlevel 1 (
    goto :process
) ELSE (
    goto :ENDER
)
:process

    REM  For Each HTML File
    REM    There is a LINK and a NAME 
    set Document_Link=%~1
    echo !Document_Link!
    set "Document_Name=!Document_Link:~0,40!"
    ECHO !Document_Name!
    
    if !Document_Link! == !Document_Name! ( 
        echo Equal 
    ) else ( 
        set Document_Name=!Document_Name!...
    )
    
Echo 1 !Document_Link!
Echo 2 !Document_Name!
    
    
:top
REM  For Each HTML
REM  BUILD Table CELL / COLUMN
    set /a count+=1
    Echo ^<td^>^&nbsp; >> !main_html!
    Echo ^<div class=^"image-container^"^> >> !main_html!
    Echo ^<a href=^">> !main_html!
REM  DYNAMIC TOP MIDDLE
    Echo !Document_Link! >> !main_html!
REM  MIDDLE
    Echo ^" target=^"_self^"^>^<img src=^"Document.jpg^" alt=^"image^"^> >> !main_html!
    ECHO ^<div class=^"overlay-text^"^> >> !main_html!
REM  DYNAMIC BOTTOM MIDDLE
Echo !Document_Name! >> !main_html!

REM  MIDDLE END
    ECHO ^</div^> >> !main_html!
    ECHO ^</a^> >> !main_html!
    ECHO ^</div^> >> !main_html!
    ECHO ^</td^> >> !main_html!

    REM  After 4 Columns ADD Table Row END
    IF "!count!"=="3" (
        Echo ^</tr^> >> !main_html!
        Echo ^<tr^> >> !main_html!
        set /a count=0
    )
goto :ENDER

:Finish
REM  Finish main.html
    Echo ^</tr^> >> !main_html!
    type "!atom_resources!BOTTOM-MAIN.txt" >> !main_html!
    goto :ENDER
    set menuitem=""
    goto :top
    ) ELSE (
      echo Try Again
      goto :top
    )
:ENDER
REM  Loop D Loop
