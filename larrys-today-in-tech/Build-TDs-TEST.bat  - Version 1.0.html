Build-TDs.bat

@echo off
Setlocal enabledelayedexpansion
set atom_resources=%~d0%~p0resources\
set cur_dir=%~d0%~p0
set file_resources=%~d0%~p0resources\FILES\*.*
set dir_resources=%~d0%~p0resources\DIRS\*.*

set main_html=!cur_dir!main.html
copy !atom_resources!*.jpg !cur_dir!
copy !atom_resources!styles.css !cur_dir!
type !atom_resources!TOP-MAIN.txt > !main_html!

set /a count=0
set CRLF=^


Echo ^<tr^>  >> !main_html!

:top
::Main Menu
set dir_file_prompt="0 = Exit!CRLF!1 = DIR!CRLF!2 = FILE!CRLF!3 = Finish!CRLF!: "
set /a count+=1
set /p menuitem=!dir_file_prompt!
REM :: Menu Choices
REM :: 0 = Exit Program
IF "!menuitem!"=="0" (
    Exit 0
) ELSE IF "!menuitem!"=="1" (

REM :: DIR BEGIN
Echo ^<td^>^&nbsp;  >> !main_html!
Echo ^<div class=^"image-container^"^>  >> !main_html!
Echo ^<a href=^" >> !main_html!
REM :: DYNAMIC TOP MIDDLE
set /p HTML_Link="Directory Link (.html): " 
Echo !HTML_Link! >> !main_html!

REM :: MIDDLE
echo ^" target=^"_top^"^>^<img src=^"Folder.jpg^" alt=^"image^"^>  >> !main_html!
echo ^<div class=^"overlay-text^"^>  >> !main_html!
REM :: DYNAMIC BOTTOM MIDDLE

set /p HTML_Link_Name="Directory Name: " 
Echo !HTML_Link_Name! >> !main_html!

REM :: DIR END
Echo ^</div^> >> !main_html!
Echo ^</a^> >> !main_html!
Echo ^</div^> >> !main_html!    
Echo ^</td^> >> !main_html!

IF "!count!"=="4" (
    Echo ^</tr^> >> !main_html!
    set /a count=0
)
rem Echo ^</tr^> >> !main_html!

goto :top
) ELSE IF "!menuitem!"=="2" (
REM :: FILES-TOP
Echo ^<td^>^&nbsp; >> !main_html!
Echo ^<div class=^"image-container^"^> >> !main_html!
Echo ^<a href=^">> !main_html!
REM :: DYNAMIC TOP MIDDLE
set /p Document_Link="Document Link (HTML): " 
Echo !Document_Link! >> !main_html!

REM :: MIDDLE
Echo ^" target=^"_self^"^>^<img src=^"Document.jpg^" alt=^"image^"^> >> !main_html!
ECHO ^<div class=^"overlay-text^"^> >> !main_html!
REM :: DYNAMIC BOTTOM MIDDLE

set /p Document_Name="Document Name: " 
Echo !Document_Name! >> !main_html!

REM :: FILES END
ECHO ^</div^> >> !main_html!
ECHO ^</a^> >> !main_html!
ECHO ^</div^> >> !main_html!
ECHO ^</td^> >> !main_html!
IF "!count!"=="4" (
    Echo ^</tr^> >> !main_html!
    set /a count=0
)
goto :top
) ELSE IF "!menuitem!"=="3" (
REM :: Finish
Echo ^</tr^> >> !main_html!
type !atom_resources!BOTTOM-MAIN.txt >> !main_html!
exit 0
goto :top
) ELSE (
  echo Try Again
  goto :top
)
