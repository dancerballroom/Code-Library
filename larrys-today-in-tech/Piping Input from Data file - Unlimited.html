2. RUN File

@echo off
Setlocal enabledelayedexpansion

:: Line by Line
:: Input Data and build the command string
:: Handles %1 through %9
for /F "tokens=*" %%A in (Pipe-Input-DATA.txt) do (
    IF "!line!"=="" (
        set line="%%A"
    ) ELSE (
        set line=!line! "%%A"
    )
)

:: Run the Batchfile using Input Data from Pipe-Input-DATA 
Call Process-ARGS.bat !line!

3. Batch file requesting USER INPUT

@echo off
Setlocal enabledelayedexpansion

:: Loop
:: Read %1 from the command line
:: Shift %n to %1
:: Possible Argument Count Limitation:  255 Characters?
set count=0
:loop
if .%1 equ . goto :exiter
set /a count=%count%+1
echo %count%:%~1

:: Assign Variables as we loop through %1 .. %n  
:: cmd1=(First Line of Data File)
:: cmd25=(25th Line of Data File)
:: cmd75=(75th Line of Data File)
set cmd%count%=%~1
shift /1
goto loop

:exiter
echo.
:: Here is where I can pre-define the input data format
:: to the needs of the current batch file.
set First_Letter=%cmd1%
set Fourth_Letter=%cmd4%
set Eighth_Letter=%cmd8%

if not defined First_Letter set /p "First Letter: "
if not defined Fourth_Letter set /p "Fourth Letter: "
if not defined Eighth_Letter set /p "Eighth Letter: "

:: Instead of Letters
:: This could be used with my Menu-Reg-Generator.bat (Link Below)
echo 1st Entry: %First_Letter%
echo 4th Entry: %Fourth_Letter%
echo 8th Entry: %Eighth_Letter%
pause
