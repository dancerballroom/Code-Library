DIR-TREE-GEN.bat

@echo off
Setlocal enabledelayedexpansion
set file_resources=%~d0%~p0resources\FILES\*.*
set dir_resources=%~d0%~p0resources\DIRS\*.*

set CRLF=^


:top
::Main Menu
set root_prompt="0 = Exit!CRLF!1 = Create Directory!CRLF!2 = Create File!CRLF!3 = To Parent or TREE!CRLF!: "

set /p menuitem=!root_prompt!
REM :: Menu Choices
REM :: 0 = Exit Program
IF "!menuitem!"=="0" (
    tree
    echo Goodbye!
    pause
    Exit 0
) ELSE IF "!menuitem!"=="1" (
    set DIR_prompt="DIR-NAME -> "
    set /p DIR-Name=!DIR_prompt!
    md "!DIR-Name!"
    pushd "!DIR-Name!"
    copy "!dir_resources!" .
    goto :top
) ELSE IF "!menuitem!"=="2" (
    REM set FILE_prompt="FILE-NAME-> "
    REM set /p File-Name=!FILE_prompt!
    copy "!file_resources!" .
    REM copy /b NUL "!FILE-Name!"
    goto :top
) ELSE IF "!menuitem!"=="3" (
    popd
    tree
    goto :top
) ELSE (
  echo Try Again
  goto :top
)
